// 2. POST - Naya product add karo
export async function POST(request) {
  try {
    const formData = await request.json();
    
    const newProduct = {
      name: formData.name,
      price: parseFloat(formData.price),
      description: formData.description,
      image: formData.image || '',
      createdAt: new Date().toISOString()
    };

    // Firebase mein save karo
    const docRef = await addDoc(productsCollection, newProduct);
    
    // Document with ID return karo
    const docSnap = await getDocs(doc(db, 'products', docRef.id));
    const product = { id: docRef.id, ...docSnap.docs[0].data() };
    
    return Response.json(product, { status: 201 });
  } catch (error) {
    console.error('POST product error:', error);
    return Response.json(
      { error: 'Product add nahi hua. Firebase error!' }, 
      { status: 500 }
    );
  }
}

// 3. DELETE - Product delete karo (bonus)
export async function DELETE(request) {
  try {
    const { id } = await request.json();
    await deleteDoc(doc(db, 'products', id));
    return Response.json({ success: true, message: 'Product delete ho gaya' });
  } catch (error) {
    return Response.json({ error: 'Delete nahi hua' }, { status: 500 });
  }
}
